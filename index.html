let trips=[]; let currentTrip; function searchTrip(){ let from=document.getElementById("from").value; let to=document.getElementById("to").value; let type=document.getElementById("type").value; if(!from||!to){ alert("Nhập đủ!"); return; } trips=[]; for(let i=0;i<6;i++){ trips.push({ from,to, time:7+i+":00", price:200+i*50, type:i%2==0?"vip":"limousine" }); } if(type!=="all"){ trips=trips.filter(t=>t.type===type); } renderTrips(); } function renderTrips(){ let html=""; trips.forEach(t=>{ html+=`

${t.from} → ${t.to}

${t.time} | ${t.type}

${t.price}k
`; }); document.getElementById("results").innerHTML=html; } function book(t){ currentTrip=t; document.getElementById("modal").style.display="block"; document.getElementById("info").innerText= `${t.from} → ${t.to} | ${t.price}k`; } function confirmBooking(){ let name=document.getElementById("name").value; let phone=document.getElementById("phone").value; if(!name||!phone){ alert("Nhập đủ!"); return; } /* lưu dữ liệu */ let data=JSON.parse(localStorage.getItem("tickets")||"[]"); data.push({...currentTrip,name,phone}); localStorage.setItem("tickets",JSON.stringify(data)); alert("Đặt vé thành công!"); closeModal(); } function closeModal(){ document.getElementById("modal").style.display="none"; }
